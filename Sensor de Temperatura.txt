#include <Wire.h>
#include <Adafruit_MLX90614.h>

Adafruit_MLX90614 mlx = Adafruit_MLX90614();

void setup() {
  Serial.begin(9600);
  Serial.println("Iniciando sensor MLX90614...");
  
  if (!mlx.begin()) {
    Serial.println("Erro ao iniciar MLX90614. Verifique conexões!");
    while (1);
  }
}

void loop() {
  float tempAmbiente = mlx.readAmbientTempC();
  float tempObjeto = mlx.readObjectTempC();

  Serial.print("Temperatura Ambiente: ");
  Serial.print(tempAmbiente);
  Serial.print(" °C\t");

  Serial.print("Temperatura Objeto: ");
  Serial.print(tempObjeto);
  Serial.println(" °C");

  delay(1000); // Aguarda 1 segundo
}
